Knoten find_min(int m){
	Knoten a_m = 0;
	Knoten tmp = root;
	Knoten last_gt = root;
	while (a_m == 0){
	  if(m == tmp.wert){
	    a_m = tmp;
	  } else {
	    if ( m < tmp.wert ){
	      last_gt = tmp;
	      if (tmp.linkesKind  == 0) am = last_gt;
	      else                      tmp = tmp.linkesKind;
	    }
	    else if (m > tmp.wert ){
	      if (tmp.rechtesKind  == 0) am = last_gt;
	      else                       tmp = tmp.rechtesKind;
	    }
	}
	return a_m;
}
